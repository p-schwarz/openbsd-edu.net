Installation Wifi

echo "deb http://ftp.fr.debian.org/debian/ wheezy main contrib non-free" >> /etc/apt/sources.list
aptitude update
aptitude install linux-headers-$(uname -r|sed 's,[^-]*-[^-]*-,,') broadcom-sta-dkms
modprobe -r b44 b43 b43legacy ssb brcmsmac
apt-get install firmware-b43-installer

Java from Sun

echo "
deb http://httpredir.debian.org/debian/ jessie main contrib
" >> /etc/apt/sources.list

apt-get update
apt-get dist-upgrade
apt-get install java-package icedtea-plugin  icedtea-netx icedtea-netx-common icedtea-7-plugin

Puis, associez les .jnlp à /usr/bin/javaws pour le navigateur.


Obsolète

Télécharger le paquet 32 ou 64 bits 

mkdir /opt/java-oracle
tar -zxf /home/USER/Téléchargements/jre-7u67-linux-x64.tar.gz -C /opt/java-oracle
JHome=/opt/java-oracle/jre1.7.0_67
mkdir /opt/java-oracle/jre1.7.0_67/bin/javac
update-alternatives --install /usr/bin/java java ${JHome%*/}/bin/java 20000
JHome=/opt/java-oracle/jre1.7.0_67
update-alternatives --install /usr/bin/javac javac ${JHome%*/}/bin/javac 20000
update-alternatives --config java
java -version

mkdir /opt/google/chrome/plugins
rm /opt/google/chrome/plugins/libnpjp2.so 
ln -s /opt/java-oracle/jre1.7.0_67/lib/amd64/libnpjp2.so /opt/google/chrome/plugins/
rm /home/USER/.mozilla/plugins/libnpjp2.so 
ln -s /opt/java-oracle/jre1.7.0_67/lib/amd64/libnpjp2.so /home/USER/.mozilla/plugins/
