pkgadd -d VirtualBox-4.1.4-SunOS-r74291.pkg
VBoxManage extpack install --replace Oracle_VM_VirtualBox_Extension_Pack-4.1.4-74291.vbox-extpack

Création d'une VM en ligne de commande (CLI)

#!/bin/bash
#Creation d'une VM en CLI
#Paramètres à modifier
TYPE=Ubuntu
RAM=1024
SIZE=80000
PATH2ISO=/home/10.iso
RIF=bge1
IPSRV_RIF=@IP

 VBoxManage createvm --name "$1" --ostype $TYPE --register
 VBoxManage modifyvm "$1" --memory $RAM --acpi on --boot1 dvd --boot2 disk  --nic1 bridged  --nictype1 82540EM --bridgeadapter1 $RIF --audio oss
  --usb on  --usbehci on --vrde on --vrdemulticon on  --vrdevideochannel on   --vrdevideochannelquality 100 --accelerate3d on --vrdeaddress $IPSRV_RIF
 VBoxManage createhd --filename "$1.vdi" --size $SIZE
 VBoxManage storagectl "$1" --name "IDE Controller" --add ide --controller PIIX4
 VBoxManage storageattach "$1" --storagectl "IDE Controller" --port 0 --device 0 --type hdd --medium "$1.vdi"
 VBoxManage storageattach "$1" --storagectl "IDE Controller" --port 0 --device 1 --type dvddrive --medium $PATH2ISO
 VBoxHeadless --startvm "$1"
  • Lancer la VM : 
VBoxHeadless --startvm VM
  • Modifier la VM pour lui intégrer une carte réseau compatible WinXP de base
VBoxManage modifyvm "VM" --nictype1 Am79C973


  • Attacher la carte réseau de la VM à la bonne interface physique du serveur avec le câble réseau virtuel branché :
VBoxManage modifyvm "VM" --nictype1 Am79C973 --bridgeadapter1 net0 --cableconnected1 on
  • Voir les infos de la VM
VBoxManage showvminfo  VM
  • Attacher un DVD au lecteur :
VBoxManage storageattach  "VM" --storagectl "IDE Controller" --port 0 --device 1 --type dvddrive --medium /root/WarcraftIII.iso :-)

Debug

Debian, pas d'accès au port USB dans la VM

Erreur d'accès au sous-système USB. Le service de relais de périphériques USB n'a pas pû être activé (erreur VERR_FILE_NOT_FOUND). Il se peut que le service ne soit pas installé sur l'ordinateur hôte.

Ajouter l'user au groupe vboxusers et redémarrer la session

  • SOuci USB de montage :

Failed to create a proxy device for the USB device. (Error: VERR_READ_ERROR).


Install Windows 7 impossible

Erreur 0xc0000225 : Il faut activer les IO Apic sur la VM.
